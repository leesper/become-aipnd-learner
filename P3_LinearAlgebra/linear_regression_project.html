def matxMultiply(A, B):
    nrowA, ncolA = shape(A)
    nrowB, ncolB = shape(B)
    if ncolA != nrowB:
        raise ValueError
    M = []
    for i in range(nrowA):
        M.append([])
        for j in range(ncolB):
            M[i].append(0)
            for k in range(nrowB):
                M[i][j] += A[i][k] * B[k][j]
    return M

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    nrowA, _ = shape(A)
    nrowb, _ = shape(b)
    if nrowA != nrowb:
        return None
    Ab = augmentMatrix(A, b)
    nrow, ncol = shape(Ab)
    for c in range(ncol-1):
        max_abs = max([abs(Ab[r][c]) for r in range(c, nrow)])
        max_row = list(filter(lambda r: abs(Ab[r][c]) == max_abs, list(range(c, nrow))))[0]
        # singular matrix
        if max_abs <= epsilon:
            return None
        swapRows(Ab, c, max_row)
        scaleRow(Ab, c, 1/Ab[c][c])
        for r in range(nrow):
            if r != c:
                addScaledRow(Ab, r, c, -Ab[r][c])
    return [[Ab[x][ncol-1]] for x in range(nrow)]
